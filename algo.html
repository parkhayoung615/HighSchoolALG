3, 4번 문제는 저작권 문제로 제공되지 않습니다.
W, N = map (int, input().split()) # 몸무게 W, 다이어트 N for _ in range(N):    c, e = map(int, input().split())    if W >= 80:      if c < e:         W -= 1    elif W <= 10:      if c > e:         W += 1   else:     if c < e:        W -= 1     elif c > e:        W += 1 print(W) 63
N = int(input()) time = 0 for _ in range(N):    t, w = map(int, input().split())    if t < time:      time += w    else:      time = t     time += w print(time) 8
